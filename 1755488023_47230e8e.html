# 优化后的可视化代码 ```python import numpy as np import matplotlib.pyplot as plt from matplotlib.animation import FuncAnimation from mpl_toolkits.mplot3d import Axes3D # 设置更美观的样式 plt.style.use('seaborn-darkgrid') # 创建图形和3D坐标轴 fig = plt.figure(figsize=(10, 8)) ax = fig.add_subplot(111, projection='3d') # 设置坐标轴标签和标题 ax.set_xlabel('X轴', fontsize=12, labelpad=10) ax.set_ylabel('Y轴', fontsize=12, labelpad=10) ax.set_zlabel('Z轴', fontsize=12, labelpad=10) ax.set_title('3D螺旋线动画', fontsize=16, pad=20) # 调整视角和边距 ax.view_init(elev=20, azim=45) plt.tight_layout() # 生成螺旋线数据 t = np.linspace(0, 20*np.pi, 1000) x = np.sin(t) * t/10 y = np.cos(t) * t/10 z = t/10 # 初始化空的线对象 line, = ax.plot([], [], [], lw=2, color='royalblue') point, = ax.plot([], [], [], 'o', color='crimson', markersize=8) # 设置坐标轴范围 ax.set_xlim(min(x), max(x)) ax.set_ylim(min(y), max(y)) ax.set_zlim(min(z), max(z)) # 添加网格和美化 ax.xaxis.pane.fill = False ax.yaxis.pane.fill = False ax.zaxis.pane.fill = False ax.xaxis.pane.set_edgecolor('w') ax.yaxis.pane.set_edgecolor('w') ax.zaxis.pane.set_edgecolor('w') ax.grid(True, linestyle='--', alpha=0.7) # 动画更新函数 def update(frame): # 更新线条 line.set_data(x[:frame], y[:frame]) line.set_3d_properties(z[:frame]) # 更新点 point.set_data([x[frame]], [y[frame]]) point.set_3d_properties([z[frame]]) # 缓慢旋转视角 ax.view_init(elev=20, azim=frame/10) return line, point # 创建动画 ani = FuncAnimation(fig, update, frames=len(t), interval=20, blit=True, repeat=True) # 添加颜色条说明 cbar = fig.colorbar(plt.cm.ScalarMappable(cmap='viridis'), ax=ax, shrink=0.5) cbar.set_label('时间进度', rotation=270, labelpad=15) plt.show() ``` ## 优化说明： 1. **视觉效果增强**： - 使用了更美观的seaborn-darkgrid样式 - 添加了3D坐标轴的网格和透明背景 - 设置了更合适的视角(elev=20, azim=45) - 增加了坐标轴标签和标题的字体大小和间距 2. **动画改进**： - 同时绘制线条和移动点，使动画更生动 - 添加了缓慢旋转的视角变化 - 使用了更鲜明的颜色(royalblue线条和crimson点) 3. **额外功能**： - 添加了颜色条表示时间进度 - 优化了图形布局和边距 - 设置了更平滑的动画间隔(20ms) 4. **代码结构**： - 分离了数据生成和动画逻辑 - 使用更清晰的变量命名 - 添加了注释说明关键步骤 这段代码将生成一个美观的3D螺旋线动画，包含线条绘制和点移动效果，同时视角会缓慢旋转，整体视觉效果更加专业和吸引人。
